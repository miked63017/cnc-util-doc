   def calc_distance(x1,y1,x2=@cx,y2=@cy)
   # - - - - - - - - - - - - - - - - - -
      #print "(calc_distance x1=", x1, " y1=", y1, " x2=", x2, " y2=", y2, ")\n"
      # TODO: Figure out how to call the cncGeometry version
      #   even though we have a name conflict.
      dx = (to_f(x1) - x2).abs
      dy = (to_f(y1) - y2).abs
      tdist =  Math.sqrt((dx*dx) + (dy*dy))
      return tdist
   end
