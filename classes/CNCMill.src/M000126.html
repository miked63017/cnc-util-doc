   def mill_rect_centered(scx,scy,width,length, depth)
   # - - - - - - - - - - - - - - - -i
     lx = to_f(scx) - (width  / 2)  
     ly = to_f(scy) - (length / 2)
     mx = to_f(scx) + (width  / 2)
     my = to_f(scy) + (length / 2)
     mill_rect(lx,ly,mx,my, depth)
   end
