   def  mill_depth(depth=nil)
   # - - - - - - - - - - - - - - - -     
     if (depth != nil)
        if (depth < @max_z)
           depth =   @max_z
        elsif (depth < @min_z)
           depth = @min_z
        end #if
        @mill_depth = depth
     end #if
     @mill_depth
   end
