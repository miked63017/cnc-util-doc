   def move_fast(xo,yo=@cy,zo=@cz)
   # - - - - - - - - - - - - - - - - 
     if (xo == @cx) && yo = @cz  && (zo == @cz)
       @no_move_count += 1
     else
       if (xo > @max_x)
         xo = @max_x
       elsif (xo < @min_x)
         xo = @min_x
       end #if
       if (yo > @max_y)
         yo = @max_y
       elsif (yo < @min_y)
         yo = @min_y
       end #if
       if (zo > @max_z)
         zo = @max_z
       elsif (zo < @min_z)
         zo = @min_z
       end #if
       print "G00 X",sprintf("%8.3f", xo),  " Y",sprintf("%8.3f", yo), " Z",sprintf("%8.3f", zo)  , " (move fast)\n"
       @cx = xo
       @cy = yo
       @cz = zo 
     end # if
   end
