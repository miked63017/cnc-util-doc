   def plung(depth = @mill_depth)
   # - - - - - - - - - - - - - - - -     
     if (mill_depth == @cz)
        @no_move_count += 1
     else
       if (depth > @max_z)
         depth = @max_z
       elsif (depth < @min_z)
         depth = @min_z
       end #if

       print "G01 Z", sprintf("%8.3f", depth), " F", @speed_plung, " (plung)\n"
       @cz = depth
     end #if
   end
