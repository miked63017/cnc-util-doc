     def base_init(mill,x=0.0,y=0.0,z=nil, depth=0.2)
     # - - - - - - - - - - - - - - - -
         if (depth == nil)
           depth = @mill.mill_depth
         end
         @mill                = mill
         @x                  = x
         @y                  = y
         if (z == nil)
           @z = @mill.cz
         else
           @z                  = z
         end
         @depth           = depth
        # print "(CNCshape.Base.initialize mill=", mill, " depth=", @depth, ")\n"
       end
