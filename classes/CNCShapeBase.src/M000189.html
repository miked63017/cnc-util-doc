       def max(n1,n2,n3=nil,n4=nil,n5=nil,n6=nil)
      # - - - - - - - - - - - - - - - -     
         keep = n1
         if (n2 > keep)
           keep = n2
         end
         if n3 != nill && n3 > keep
           keep = n3
         end
         if n4 != nil && n4 > keep
           keep = n4
         end
         if n5 != nil && n5 > keep
           keep = n5
         end
         if n6 != nil && n6 > keep
           keep = n6
         end
        @keep
       end
