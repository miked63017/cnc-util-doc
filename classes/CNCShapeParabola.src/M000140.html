    def initialize(mill, width, focus_y, depth,x_increment=0.1)
    # - - - - - - - - - - - - - - - -    
      base_init(mill)
      @mill         = mill
      @width      = width
      @focus_y   = focus_y
      @depth      = depth
      @y_off       = 0.3
      half_width  = width / 2
      @min_x      = 0 - half_width
      @max_x     = half_width
      @min_y      = @y_off
      @max_y     = @y_off + calc_y(max_x, focus_y)
      @side_off   = 1.0
      @x_increment = x_increment
    end
