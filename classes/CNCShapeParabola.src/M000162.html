    def width(aWidth=nil)
    # - - - - - - - - - - - - - - - -
      if (aWidth != nil)
        @width = aWidth
        half_width = aWidth / 2
        @max_x = half_width
        @min_x  = 0 - half_width
        @max_y   = depth_off + calc_y(@max_x, @focus_point)
      end
      @width
    end
