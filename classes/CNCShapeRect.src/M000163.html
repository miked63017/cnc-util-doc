      def centered(scx=@x,scy=@y,width=@width,length=@len, depth=@depth)
      # - - - - - - - - - - - - - - - -  
        #print "(centered scx=", scx, " scy=", scy, " width=", width, " length = ", length,  " depth= ", depth, ")\n"
        @width = width.abs
        @len     = length.abs
        @depth =depth
        wr = width / 2
        lr  = length / 2
        @lx       = scx - wr  
        @ly       = scy - lr
        @mx     = scx + wr
        @my     = scy + lr
        self
      end
