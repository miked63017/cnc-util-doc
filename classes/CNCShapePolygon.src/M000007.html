  def do_mill
  # - - - - - - - - - - - - -
    target_depth = @depth
    if depth <= mill.cut_depth_inc_curr
      do_mill_s(target_depth)
    else
      # Need multiple passes
      tdepth = mill.cut_depth_inc_curr
      while true
        do_mill_s(tdepth)
        if (tdepth == target_depth)
           break # have reached target depth
        end #if
        tdepth += mill.cut_depth_inc_curr
        if (tdepth > target_depth)
          # want to make sure final cut doesn't
          # go past tarteget.
          tdepth = target_depth
        end #if
      end #while
    end #else

  end
