def mill_collet_slot(mill, x,y, bore_diam)
##################
  aCircle = CNCShapeCircle.new(mill)

  # mill out main pocket
  aCircle.mill_pocket(x = x, 
                      y = y, 
                      diam = C_out_diam, 
                      depth= C_depth,  
                      island_diam=bore_diam)

  # mill out bottom pocket
  aCircle.mill_pocket(x = x, y = y, 
                      diam = C_bot_diam, 
                      depth = C_bot_depth,  
                      island_diam = bore_diam)
  mill.retract()

end
