def calc_angle(x1, y1,  x2, y2)
# - - - - - - - - - - - - - - - - - - - -
  if (x1 == x2)
    if (y2 >= y1)
      return 0
    else
      return 180
    end #else
  elsif (y1 == y2)
    if (x2 >= x1)
      return 90
    else
      return 270
    end #else
  else
    x1 = x1 + 0.0 # force to be floating point
    x2 = x2 + 0.0 # force to floating point
    slope =  (y2-y1) / (x2-x1)
    radians =   Math.atan(slope)
    degrees = conv_radian_to_degree(radians)
    #  Handle Adjusting Quadrent
    if ((x2 > x1) && (y2 > y1))
      # Quadrent 1;
      degrees =  degrees
    elsif ((x2 > x1) && (y2 < y1))
     # Quadrent 2
     degrees = 90 - degrees;
    elsif ((x2 < x1) && (y2 < y1))
     # Quadrent 3
     degrees = 270 - degrees
    else
     # must be Quadrent 4
     degrees = 270 - degrees
    end # else
  end # else
  return degrees
end
