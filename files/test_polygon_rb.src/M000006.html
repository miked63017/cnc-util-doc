def test_octagon_pocket
######################################
  aMill = CNCMill.new
  aMill.job_start()
  aMill.home()
  #  variables used here to make
  #  easy for reader.
  aPoly =  CNCShapePolygon.new(aMill,  
    cent_x = 2.5,  cent_y = 1.5,        
    start_z = 0.0, diam=1.5,  num_sides=6,
    depth=-1.2,  degree_inc=nil)
  aPoly.do_mill()
  aMill.job_finish()
end
